// Скопируй это в Firebase Console → Firestore Database → Rules → и нажми "Publish"
// Замени ВЕСЬ текст в редакторе на этот.
//
// UID администратора уже вставлен ниже — если захочешь сменить админ-аккаунт в будущем,
// не забудь заменить его в функции isAdmin() перед публикацией.
//
// Модель доступа с этой версии:
// - Читать рецепты и отзывы может кто угодно с авторизацией (в т.ч. анонимной — она происходит
//   автоматически при открытии сайта).
// - Создавать, полностью редактировать и удалять рецепты, а также удалять/редактировать чужие
//   отзывы — может ТОЛЬКО автор (тот, кто вошёл по паролю, см. admin-auth.js).
// - Любой посетитель (даже анонимный) по-прежнему может: поставить/снять избранное, отметить
//   "буду готовить", добавить фото к рецепту, оставить свой отзыв — эти действия не требуют пароля,
//   это осознанное решение, чтобы близкие могли пользоваться книгой без входа.

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    function isAdmin() {
      return request.auth != null && request.auth.uid == '4XwOCBBWfgcYNjPHJogPxzEmrt23';
    }

    // ---------- Рецепт: проверка формы и границ (для полного создания/редактирования автором) ----------
    function isValidRecipe(data) {
      return data.title is string && data.title.size() > 0 && data.title.size() <= 200
        && data.category is string && data.category.size() <= 60
        && (!('servings' in data) || data.servings == null || (data.servings is number && data.servings > 0 && data.servings <= 100))
        && (!('cookTime' in data) || data.cookTime == null || (data.cookTime is number && data.cookTime > 0 && data.cookTime <= 1440))
        && (!('difficulty' in data) || data.difficulty in ['Легко','Средне','Сложно'])
        && data.ingredients is list && data.ingredients.size() <= 60
        && data.steps is list && data.steps.size() <= 60
        && (!('notes' in data) || data.notes == null || (data.notes is string && data.notes.size() <= 2000))
        && (!('favorite' in data) || data.favorite is bool)
        && (!('willCook' in data) || data.willCook is bool)
        && (!('photos' in data) || (data.photos is list && data.photos.size() <= 20));
    }

    // ---------- Отзыв: оценка обязана быть числом от 1 до 5 ----------
    function isValidReview(data) {
      return data.rating is number && data.rating >= 1 && data.rating <= 5
        && (!('text' in data) || data.text == null || (data.text is string && data.text.size() <= 1000))
        && data.createdAt is string;
    }

    match /recipes/{recipeId} {
      allow read: if request.auth != null;

      // Полное создание и редактирование рецепта — только автор.
      allow create: if isAdmin() && isValidRecipe(request.resource.data);

      // Обновление: либо автор меняет рецепт целиком (с проверкой формы),
      // либо кто угодно авторизованный меняет ТОЛЬКО избранное/"буду готовить"/фото —
      // остальные поля при этом трогать нельзя.
      allow update: if (isAdmin() && isValidRecipe(request.resource.data))
        || (
          request.auth != null
          && request.resource.data.diff(resource.data).affectedKeys().hasOnly(['favorite','willCook','photos'])
          && (!('photos' in request.resource.data) || (request.resource.data.photos is list && request.resource.data.photos.size() <= 20))
        );

      allow delete: if isAdmin();

      match /reviews/{reviewId} {
        allow read: if request.auth != null;
        // Оставить отзыв может кто угодно авторизованный (с проверкой формы).
        allow create: if request.auth != null && isValidReview(request.resource.data);
        // Редактировать/удалять чужие отзывы — только автор (модерация).
        allow update, delete: if isAdmin();
      }
    }

    match /meta/{document} {
      allow read, write: if request.auth != null;
    }
  }
}
